Dim val As String
private sub BINARY_click()
        Dim tmp As String
        Dim num As Double
        Dim n As Double
        num = Text1.Text
        n = num
        tmp = Trim(str(n mod 2))
        n = n/2
        Do While n<>0
        tmp = Trim(str(n mod 2)) & tmp
        n = n/2
        Loop 
        num = tmp
        Text2.Text = str(tmp)
End sub

        End
End sub
